

async reopen(): Promise<Page> {
        const url = this.page.url();
        const context = this.page.context();

        await this.page.close();

        const newPage = await context.newPage();
        await newPage.goto(url);

        return newPage;
    }

async choseOptionfromSidebar(value:string):Promise<void>{

        const count =  this.sideBarList.count();

        for(let i = 0; i < await count; i++){
        const options = this.sideBarList.nth(i);
        const title = await options.textContent();
        if(title === value){
            await options.click();
            break;
        }

        }
    
    }

    sync pressTabKey(email: string): Promise<void> {


        const emailField = this.emailInputField;

        for (let i = 0; i < 50; i++) {
            await this.page.keyboard.press('Tab');

            try {
                await expect(emailField).toBeFocused({ timeout: 200 });
                await emailField.fill(email);
                return; // success → exit function
            } catch (e) {
                // ignore and continue tabbing
            }
        }

        // final assertion (will fail the test)
        await expect(emailField).toBeFocused();

    }
